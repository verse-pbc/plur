Onboarding flow pseudo-code:
```javascript
const onboardUser = async () => {
  // Generate keypair
  const keypair = generateKeypair();
  await storeKeypair(keypair);
  
  // Generate pseudonym
  const pseudonym = generatePseudonym();
  
  // Initialize Cashu wallet
  const wallet = await initializeCashuWallet();
  
  // Create user profile
  const profile = {
    displayName: pseudonym,
    tags: [],
    walletPublic: false,
    createdAt: new Date().toISOString()
  };
  
  await storeUserProfile(profile);
  
  return { keypair, profile, wallet };
};
```
