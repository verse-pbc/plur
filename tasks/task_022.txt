Off-ramp implementation:
```javascript
const exportUserData = async () => {
  const userData = await getUserData();
  const relays = await getRelayList();
  
  // Format nsec for export
  const nsec = nip19.nsecEncode(userData.privateKey);
  
  // Format relay list
  const relayList = relays.map(relay => ({
    url: relay.url,
    read: relay.read,
    write: relay.write
  }));
  
  return {
    nsec,
    npub: userData.npub,
    relays: relayList
  };
};

const generateExportQR = async () => {
  const data = await exportUserData();
  
  // Create JSON string for QR
  const qrData = JSON.stringify({
    privateKey: data.nsec,
    relays: data.relays
  });
  
  // Generate QR code
  return generateQRCode(qrData);
};
```
